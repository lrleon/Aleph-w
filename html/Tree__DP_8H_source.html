76 template <class GT, class SA>
77 class Tree_Topology
78 {
79 public:
80 using Node = typename GT::Node;
81 using Arc = typename GT::Arc;
84 static constexpr size_t NONE = std::numeric_limits<size_t>::max();
85
86 private:
87 const GT *graph_ = nullptr;
88 SA sa_;
89 Node *root_ = nullptr;
90 size_t n_ = 0;
92 Array<Node *> id_to_node_;
93 MapOLhash<Node *, size_t> node_to_id_;
94 Array<Array<size_t>> adj_;
95 Array<size_t> parent_;
96 Array<size_t> order_;
99 void index_nodes()
100 {
101 n_ = graph_->get_num_nodes();
102 if (n_ == 0)
103 {
104 ah_domain_error_if(root_ != nullptr)
105 << "Tree_Topology: non-null root provided for empty graph";
106 return;
107 }
108
109 id_to_node_ = Array<Node *>::create(n_); {
110 MapOLhash<Node *, size_t> tmp(n_);
111 node_to_id_.swap(tmp);
112 }
113
114 size_t next_id = 0;
115 for (Node_Iterator<GT> it(*graph_); it.has_curr(); it.next_ne())
116 {
117 Node *p = it.get_curr();
118 id_to_node_(next_id) = p;
119 node_to_id_.insert(p, next_id);
120 ++next_id;
121 }
122
123 if (root_ == nullptr)
124 root_ = id_to_node_(0);
125 else
126 ah_domain_error_if(node_to_id_.search(root_) == nullptr)
127 << "Tree_Topology: root node is not part of the graph";
128 }
129
131 void build_adjacency()
132 {
133 if (n_ == 0)
134 return;
135
136 adj_.reserve(n_);
137 for (size_t i = 0; i < n_; ++i)
138 adj_.append(Array<size_t>());
139
140 using Pair_Key = std::pair<size_t, size_t>;
141 DynMapTree<Pair_Key, char> unique_edges;
142 size_t edge_count = 0;
143
144 for (Arc_Iterator<GT, SA> it(*graph_, sa_); it.has_curr(); it.next_ne())
145 {
146 Arc *a = it.get_curr_ne();
147 Node *src = graph_->get_src_node(a);
148 Node *tgt = graph_->get_tgt_node(a);
149
150 const auto *si = node_to_id_.search(src);
151 const auto *ti = node_to_id_.search(tgt);
152 ah_runtime_error_unless(si != nullptr and ti != nullptr)
153 << "Tree_Topology: arc endpoint not indexed";
154
155 const size_t u = si->second;
156 const size_t v = ti->second;
157 ah_domain_error_if(u == v)
158 << "Tree_Topology: self-loop detected";
159
160 Pair_Key key = u < v ? std::make_pair(u, v) : std::make_pair(v, u);
161 if (unique_edges.search(key) != nullptr)
162 continue; // skip duplicate
163 unique_edges.insert(key, 1);
164 ++edge_count;
165 }
166
167 ah_domain_error_if(n_ > 1 and edge_count != n_ - 1)
168 << "Tree_Topology: not a tree (expected " << (n_ - 1)
169 << " edges, got " << edge_count << ")";
170
171 for (typename DynMapTree<Pair_Key, char>::Iterator it(unique_edges);
172 it.has_curr(); it.next_ne())
173 {
174 const auto & [fst, snd] = it.get_curr();
175 const size_t u = fst.first;
176 const size_t v = fst.second;
177 adj_(u).append(v);
178 adj_(v).append(u);
179 }
180 }
181
183 void build_order()
184 {
185 if (n_ == 0)
186 return;
187
188 parent_ = Array<size_t>::create(n_);
189 for (size_t i = 0; i < n_; ++i)
190 parent_(i) = NONE;
191
192 order_.reserve(n_);
193
194 Array<char> visited = Array<char>::create(n_);
195 for (size_t i = 0; i < n_; ++i)
196 visited(i) = 0;
197
198 const size_t root_id = node_to_id_.find(root_);
199
200 struct Frame
201 {
202 size_t node;
203 size_t next_child;
204 };
205 DynListStack<Frame> stack;
206
207 visited(root_id) = 1;
208 stack.push({root_id, 0});
209
210 while (not stack.is_empty())
211 {
212 auto & fr = stack.top();
213 if (fr.next_child == adj_(fr.node).size())
214 {
215 order_.append(fr.node);
216 (void) stack.pop();
217 continue;
218 }
219
220 const size_t nxt = adj_(fr.node)(fr.next_child++);
221 if (visited(nxt))
222 continue;
223
224 visited(nxt) = 1;
225 parent_(nxt) = fr.node;
226 stack.push({nxt, 0});
227 }
228
229 ah_domain_error_if(order_.size() != n_)
230 << "Tree_Topology: graph is not connected";
231 }
232
233 public:
240 Tree_Topology(const GT & g, Node *root, SA sa = SA())
241 : graph_(&g), sa_(std::move(sa)), root_(root)
242 {
243 index_nodes();
244 build_adjacency();
245 build_order();
246 }
247
249 [[nodiscard]] size_t size() const noexcept { return n_; }
250
252 [[nodiscard]] Node * root() const noexcept { return root_; }
253
255 [[nodiscard]] size_t id_of(Node *node) const
256 {
257 const auto *item = node_to_id_.search(node);
258 ah_domain_error_if(item == nullptr)
259 << "Tree_Topology::id_of: node not in graph";
260 return item->second;
261 }
262
264 [[nodiscard]] Node * node_of(size_t id) const
265 {
266 ah_out_of_range_error_if(id >= n_)
267 << "Tree_Topology::node_of: id out of range";
268 return id_to_node_(id);
269 }
270
272 [[nodiscard]] const Array<size_t> &children(size_t id) const
273 {
274 return adj_(id);
275 }
276
278 [[nodiscard]] size_t parent(size_t id) const noexcept
279 {
280 return parent_(id);
281 }
282
284 [[nodiscard]] const Array<size_t> &post_order() const noexcept
285 {
286 return order_;
287 }
288
290 [[nodiscard]] const Array<Node *> &nodes() const noexcept
291 {
292 return id_to_node_;
293 }
294 };

320 template <class GT, typename T, class SA = Dft_Show_Arc<GT>>
321 class Gen_Tree_DP
322 {
323 public:
324 using Node = typename GT::Node;
327 using Init_Fn = std::function<T(Node *)>;
328
334 using Combine_Fn = std::function<T(Node *, const T &, Node *, const T &)>;
335
336 private:
337 tree_dp_detail::Tree_Topology<GT, SA> topo_;
338 Array<T> dp_;
340 public:
349 Gen_Tree_DP(const GT & g, Node *root,
350 Init_Fn init, Combine_Fn combine, SA sa = SA())
351 : topo_(g, root, std::move(sa))
352 {
353 const size_t n = topo_.size();
354 if (n == 0)
355 return;
356
357 dp_ = Array<T>::create(n);
358
359 // Initialize all nodes
360 for (size_t i = 0; i < n; ++i)
361 dp_(i) = init(topo_.node_of(i));
362
363 // Process in post-order (leaves first)
364 const auto & order = topo_.post_order();
365 for (size_t k = 0; k < n; ++k)
366 {
367 const size_t v = order[k];
368 const size_t par = topo_.parent(v);
369 if (par == tree_dp_detail::Tree_Topology<GT, SA>::NONE)
370 continue; // root, no parent to update
371 dp_(par) = combine(topo_.node_of(par), dp_(par),
372 topo_.node_of(v), dp_(v));
373 }
374 }
375
380 [[nodiscard]] const T &value(Node *node) const
381 {
382 return dp_(topo_.id_of(node));
383 }
384
386 [[nodiscard]] const Array<T> &values() const noexcept
387 {
388 return dp_;
389 }
390
392 [[nodiscard]] size_t size() const noexcept
393 {
394 return topo_.size();
395 }
396
401 [[nodiscard]] Node * node_of(size_t id) const
402 {
403 return topo_.node_of(id);
404 }
405
410 [[nodiscard]] size_t id_of(Node *node) const
411 {
412 return topo_.id_of(node);
413 }
414 };

449 template <class GT, typename T, class SA = Dft_Show_Arc<GT>>
450 class Gen_Reroot_DP
451 {
452 public:
453 using Node = typename GT::Node;
456 using Init_Fn = std::function<T(Node *)>;
457
459 using Merge_Fn = std::function<T(const T &, const T &)>;
460
466 using Apply_Edge_Fn = std::function<T(Node *, Node *, const T &)>;
467
468 private:
469 tree_dp_detail::Tree_Topology<GT, SA> topo_;
470 T identity_;
471 Array<T> dp_down_;
472 Array<T> dp_up_;
473 Array<T> answer_;
475 public:
486 Gen_Reroot_DP(const GT & g, Node *root,
487 const T & identity, Init_Fn init,
488 Merge_Fn merge, Apply_Edge_Fn apply_edge,
489 SA sa = SA())
490 : topo_(g, root, std::move(sa)), identity_(identity)
491 {
492 const size_t n = topo_.size();
493 if (n == 0)
494 return;
495
496 dp_down_ = Array<T>::create(n);
497 dp_up_ = Array<T>::create(n);
498 answer_ = Array<T>::create(n);
499
500 for (size_t i = 0; i < n; ++i)
501 {
502 dp_down_(i) = init(topo_.node_of(i));
503 dp_up_(i) = identity_;
504 }
505
506 // Phase 1: bottom-up
507 const auto & order = topo_.post_order();
508 for (size_t k = 0; k < n; ++k)
509 {
510 const size_t v = order[k];
511 const size_t par = topo_.parent(v);
512 if (par == tree_dp_detail::Tree_Topology<GT, SA>::NONE)
513 continue;
514 const T contrib = apply_edge(topo_.node_of(par),
515 topo_.node_of(v), dp_down_(v));
516 dp_down_(par) = merge(dp_down_(par), contrib);
517 }
518
519 // Phase 2: top-down with prefix/suffix
520 // Process nodes in reverse post-order (root first)
521 for (size_t k = n; k > 0; --k)
522 {
523 const size_t v = order[k - 1];
524 Node *vn = topo_.node_of(v);
525
526 // Collect children of v (neighbors except parent)
527 const auto & adj = topo_.children(v);
528 Array<size_t> children;
529 for (size_t j = 0; j < adj.size(); ++j)
530 if (adj[j] != topo_.parent(v))
531 children.append(adj[j]);
532
533 const size_t nc = children.size();
534 if (nc == 0)
535 continue;
536
537 Array<T> contribs = Array<T>::create(nc);
538 for (size_t j = 0; j < nc; ++j)
539 {
540 const size_t c = children[j];
541 contribs(j) = apply_edge(vn, topo_.node_of(c), dp_down_(c));
542 }
543
544 // Build prefix and suffix arrays of merged contributions
545 Array<T> prefix = Array<T>::create(nc + 1);
546 Array<T> suffix = Array<T>::create(nc + 1);
547 prefix(0) = identity_;
548 suffix(nc) = identity_;
549
550 for (size_t j = 0; j < nc; ++j)
551 prefix(j + 1) = merge(prefix(j), contribs(j));
552
553 for (size_t j = nc; j > 0; --j)
554 suffix(j - 1) = merge(suffix(j), contribs(j - 1));
555
556 const T base = merge(init(vn), dp_up_(v));
557
558 // For each child, compute dp_up
559 for (size_t j = 0; j < nc; ++j)
560 {
561 const size_t c = children[j];
562 Node *cn = topo_.node_of(c);
563 // Value of v without child c's subtree:
564 // merge(init(v), dp_up(v), prefix(j), suffix(j+1))
565 T without_c = merge(base, prefix(j));
566 without_c = merge(without_c, suffix(j + 1));
567 dp_up_(c) = apply_edge(cn, vn, without_c);
568 }
569 }
570
571 // Phase 3: compute answers
572 for (size_t i = 0; i < n; ++i)
573 answer_(i) = merge(dp_down_(i), dp_up_(i));
574 }
575
580 [[nodiscard]] const T &value(Node *node) const
581 {
582 return answer_(topo_.id_of(node));
583 }
584
586 [[nodiscard]] const Array<T> &values() const noexcept
587 {
588 return answer_;
589 }
590
592 [[nodiscard]] size_t size() const noexcept
593 {
594 return topo_.size();
595 }
596
601 [[nodiscard]] Node * node_of(size_t id) const
602 {
603 return topo_.node_of(id);
604 }
605
610 [[nodiscard]] size_t id_of(Node *node) const
611 {
612 return topo_.id_of(node);
613 }

636 template <class GT, class SA = Dft_Show_Arc<GT>>
637 [[nodiscard]] Array<size_t>
638 tree_subtree_sizes(const GT & g, typename GT::Node *root, SA sa = SA())
639 {
640 Gen_Tree_DP<GT, size_t, SA> dp(g, root,
641 [](auto *) -> size_t { return 1; },
642 [](auto *, const size_t & acc, auto *, const size_t & child) -> size_t
643 {
644 return acc + child;
645 },
646 std::move(sa));
647
648 // Copy values out
649 const auto & vals = dp.values();
650 Array<size_t> result = Array<size_t>::create(vals.size());
651 for (size_t i = 0; i < vals.size(); ++i)
652 result(i) = vals[i];
653 return result;
654 }

701 template <class GT, class SA = Dft_Show_Arc<GT>>
702 [[nodiscard]] Array<size_t>
703 tree_sum_of_distances(const GT & g, typename GT::Node *root, SA sa = SA())
704 {
705 using P = std::pair<size_t, size_t>; // (count, sum_dist)
706
707 Gen_Reroot_DP<GT, P, SA> dp(g, root,

709 [](auto *) -> P { return {1, 0}; },
710 [](const P & a, const P & b) -> P
711 {
712 return {a.first + b.first, a.second + b.second};
713 },
714 [](auto *, auto *, const P & v) -> P
715 {
716 return {v.first, v.second + v.first};
717 },
718 std::move(sa));
719
720 const auto & vals = dp.values();
721 Array<size_t> result = Array<size_t>::create(vals.size());
722 for (size_t i = 0; i < vals.size(); ++i)
