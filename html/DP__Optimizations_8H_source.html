121 template <typename Cost, class Transition_Cost_Fn>
122 [[nodiscard]] inline Divide_Conquer_DP_Result<Cost>
123 divide_and_conquer_partition_dp(const size_t groups,
124 const size_t n,
125 Transition_Cost_Fn transition_cost,
126 const Cost inf =
127 dp_optimization_detail::default_inf<Cost>())
128 {
129 Array<Cost> prev = Array<Cost>::create(n + 1);
130 Array<Cost> curr = Array<Cost>::create(n + 1);
131
132 for (size_t i = 0; i <= n; ++i)
133 {
134 prev(i) = inf;
135 curr(i) = inf;
136 }
137 prev(0) = Cost{};
138
139 Array<Array<size_t>> split;
140 split.reserve(groups + 1);
141 for (size_t g = 0; g <= groups; ++g)
142 {
143 Array<size_t> row = Array<size_t>::create(n + 1);
144 for (size_t i = 0; i <= n; ++i)
145 row(i) = 0;
146 split.append(std::move(row));
147 }
148
149 for (size_t g = 1; g <= groups; ++g)
150 {
151 for (size_t i = 0; i <= n; ++i)
152 curr(i) = inf;
153
154 if (g <= n)
155 {
156 auto solve = [&](auto && self,
157 const size_t left, const size_t right,
158 const size_t opt_left, const size_t opt_right) -> void
159 {
160 if (left > right)
161 return;
162
163 const size_t mid = std::midpoint(left, right);
164 const size_t k_end = std::min(mid - 1, opt_right);
165
166 Cost best = inf;
167 size_t best_k = opt_left;
168
169 for (size_t k = opt_left; k <= k_end; ++k)
170 {
171 if (prev[k] >= inf)
172 continue;
173
174 const Cost tc = transition_cost(k, mid);
175 Cost cand;
176 if (tc >= inf or prev[k] > inf - tc)
177 cand = inf;
178 else
179 cand = prev[k] + tc;
180
181 if (cand < best)
182 {
183 best = cand;
184 best_k = k;
185 }
186 }
187
188 curr(mid) = best;
189 split[g](mid) = best_k;
190
191 if (left < mid)
192 self(self, left, mid - 1, opt_left, best_k);
193 if (mid < right)
194 self(self, mid + 1, right, best_k, opt_right);
195 };
196
197 solve(solve, g, n, g - 1, n - 1);
198 }
199
200 prev.swap(curr);
201 }
202
203 return Divide_Conquer_DP_Result<Cost>{
204 prev[n],
205 std::move(prev),
206 std::move(split)
207 };
208 }

244 template <typename Cost, class Interval_Cost_Fn>
245 [[nodiscard]] inline Knuth_Optimization_Result<Cost>
246 knuth_optimize_interval(const size_t n,
247 Interval_Cost_Fn interval_cost,
248 const Cost inf =
249 dp_optimization_detail::default_inf<Cost>())
250 {
251 Array<Array<Cost>> dp;
252 dp.reserve(n + 1);
253 for (size_t i = 0; i <= n; ++i)
254 {
255 Array<Cost> row = Array<Cost>::create(n + 1);
256 for (size_t j = 0; j <= n; ++j)
257 row(j) = inf;
258 dp.append(std::move(row));
259 }
260
261 Array<Array<size_t>> opt;
262 opt.reserve(n + 1);
263 for (size_t i = 0; i <= n; ++i)
264 {
265 Array<size_t> row = Array<size_t>::create(n + 1);
266 for (size_t j = 0; j <= n; ++j)
267 row(j) = 0;
268 opt.append(std::move(row));
269 }
270
271 for (size_t i = 0; i <= n; ++i)
272 {
273 dp(i)(i) = Cost{};
274 opt(i)(i) = i;
275 if (i + 1 <= n)
276 {
277 dp(i)(i + 1) = Cost{};
278 opt(i)(i + 1) = i + 1;
279 }
280 }
281
282 for (size_t len = 2; len <= n; ++len)
283 for (size_t i = 0; i + len <= n; ++i)
284 {
285 const size_t j = i + len;
286
287 size_t left = opt[i][j - 1];
288 size_t right = opt[i + 1][j];
289 if (left < i + 1)
290 left = i + 1;
291 if (right > j - 1)
292 right = j - 1;
293
294 ah_runtime_error_if(left > right)
295 << "knuth_optimize_interval: invalid opt bounds at ["
296 << i << ", " << j << ")";
297
298 const Cost w = interval_cost(i, j);
299
300 Cost best = inf;
301 size_t best_k = left;
302
303 for (size_t k = left; k <= right; ++k)
304 {
305 const Cost d1 = dp[i][k];
306 const Cost d2 = dp[k][j];
307 Cost cand;
308 if (d1 >= inf or d2 >= inf or w >= inf or d1 > inf - d2 or (d1 + d2) > inf - w)
309 cand = inf;
310 else
311 cand = d1 + d2 + w;
312
313 if (cand < best)
314 {
315 best = cand;
316 best_k = k;
317 }
318 }
319
320 dp(i)(j) = best;
321 opt(i)(j) = best_k;
322 }
323
324 return Knuth_Optimization_Result<Cost>{
325 n == 0 ? Cost{} : dp[0][n],
326 std::move(dp),
327 std::move(opt)
328 };
329 }

342 [[nodiscard]] inline Knuth_Optimization_Result<size_t>
343 optimal_merge_knuth(const Array<size_t> & weights)
344 {
345 const size_t n = weights.size();
346 Array<size_t> prefix = Array<size_t>::create(n + 1);
347 prefix(0) = 0;
348
349 for (size_t i = 0; i < n; ++i)
350 {
351 ah_runtime_error_if(prefix[i] >
352 std::numeric_limits<size_t>::max() - weights[i])
353 << "optimal_merge_knuth: prefix sum overflow";
354 prefix(i + 1) = prefix[i] + weights[i];
355 }
356
357 return knuth_optimize_interval<size_t>(
358 n,
359 [&](const size_t i, const size_t j) -> size_t
360 {
361 return prefix[j] - prefix[i];
362 },
363 std::numeric_limits<size_t>::max() / 4
364 );
365 }

376 template <typename T>
377 class Convex_Hull_Trick
378 {
379 static_assert(std::is_signed_v<T>, "Convex_Hull_Trick requires a signed type");
380
381 public:
383 struct Line
384 {
385 T slope = T{};
386 T intercept = T{};
387
388 [[nodiscard]] T value_at(const T x) const noexcept
389 {
390 using PromotedT = dp_optimization_detail::promoted_t<T>;
391 return static_cast<T>(static_cast<PromotedT>(slope) * static_cast<PromotedT>(x) +
392 static_cast<PromotedT>(intercept));
393 }
394 };
395
396 private:
397 Array<Line> lines_;
398 Array<long double> starts_; // x where this line becomes optimal
399 size_t cursor_ = 0; // for monotone queries
400
401 [[nodiscard]] static long double intersection_x(const Line & a,
402 const Line & b) noexcept
403 {
404 return static_cast<long double>(b.intercept - a.intercept)
405 / static_cast<long double>(a.slope - b.slope);
406 }
407
408 public:
409 [[nodiscard]] size_t size() const noexcept { return lines_.size(); }
410 [[nodiscard]] bool is_empty() const noexcept { return lines_.size() == 0; }
411
412 void clear()
413 {
414 Array<Line> tmp_lines;
415 lines_.swap(tmp_lines);
416
417 Array<long double> tmp_starts;
418 starts_.swap(tmp_starts);
419
420 cursor_ = 0;
421 }
422
423 void reset_query_cursor() noexcept
424 {
425 cursor_ = 0;
426 }
427
429 void add_line(const T slope, const T intercept)
430 {
431 Line ln{slope, intercept};
432
433 if (lines_.size() > 0)
434 {
435 const Line & last = lines_[lines_.size() - 1];
436 ah_domain_error_if(ln.slope > last.slope)
437 << "Convex_Hull_Trick::add_line: slopes must be non-increasing";
438
439 // Keep only best intercept for equal slope.
440 if (ln.slope == last.slope)
441 {
442 if (ln.intercept >= last.intercept)
443 return;
444
445 (void) lines_.remove_last();
446 (void) starts_.remove_last();
447 if (cursor_ > lines_.size())
448 cursor_ = lines_.size();
449 }
450 }
451
452 while (lines_.size() > 0)
453 {
454 const long double x = intersection_x(lines_[lines_.size() - 1], ln);
455
456 if (lines_.size() == 1 or x > starts_[starts_.size() - 1])
457 {
458 starts_.append(x);
459 lines_.append(ln);
460 if (cursor_ >= lines_.size())
461 cursor_ = lines_.size() - 1;
462 return;
463 }
464
465 (void) lines_.remove_last();
466 (void) starts_.remove_last();
467 if (cursor_ > lines_.size())
468 cursor_ = lines_.size();
469 }
470
471 starts_.append(-std::numeric_limits<long double>::infinity());
472 lines_.append(ln);
473 if (cursor_ >= lines_.size())
474 cursor_ = lines_.size() - 1;
475 }
476
478 [[nodiscard]] T query(const T x) const
479 {
480 ah_runtime_error_if(lines_.size() == 0)
481 << "Convex_Hull_Trick::query: no lines available";
482
483 size_t lo = 0;
484 size_t hi = lines_.size() - 1;
485 const auto xd = static_cast<long double>(x);
486
487 while (lo < hi)
488 if (const size_t mid = std::midpoint(lo, hi + 1); starts_[mid] <= xd)
489 lo = mid;
490 else
491 hi = mid - 1;
492
493 return lines_[lo].value_at(x);
494 }
495
497 [[nodiscard]] T query_monotone(const T x)
498 {
499 ah_runtime_error_if(lines_.size() == 0)
500 << "Convex_Hull_Trick::query_monotone: no lines available";
501
502 while (cursor_ + 1 < lines_.size()
503 and starts_[cursor_ + 1] <= static_cast<long double>(x))
504 ++cursor_;
505
506 return lines_[cursor_].value_at(x);
507 }
508 };

518 template <typename T>
519 class Li_Chao_Tree
520 {
521 static_assert(std::is_integral_v<T> and std::is_signed_v<T>,
522 "Li_Chao_Tree requires a signed integral coordinate/value type");
523
524 public:
526 struct Line
527 {
528 T slope = T{};
529 T intercept = T{};
530
531 [[nodiscard]] T value_at(const T x) const noexcept
532 {
533 using PromotedT = dp_optimization_detail::promoted_t<T>;
534 return static_cast<T>(static_cast<PromotedT>(slope) * static_cast<PromotedT>(x) +
535 static_cast<PromotedT>(intercept));
536 }
537 };
538
539 private:
540 struct Node
541 {
542 Line line;
543 size_t left = std::numeric_limits<size_t>::max();
544 size_t right = std::numeric_limits<size_t>::max();
545 };
546
547 static constexpr size_t NIL = std::numeric_limits<size_t>::max();
548
549 T x_left_;
550 T x_right_;
551 Array<Node> nodes_;
552 size_t root_ = NIL;
553
554 [[nodiscard]] size_t new_node(const Line & line)
555 {
556 nodes_.append(Node{line, NIL, NIL});
557 return nodes_.size() - 1;
558 }
559
560 void add_line_impl(const size_t idx, const T l, const T r, Line line)
561 {
562 const T mid = std::midpoint(l, r);
563
564 if (line.value_at(mid) < nodes_[idx].line.value_at(mid))
565 std::swap(line, nodes_[idx].line);
566
567 if (l == r)
568 return;
569
570 if (line.value_at(l) < nodes_[idx].line.value_at(l))
571 {
572 if (nodes_[idx].left == NIL)
573 {
574 nodes_[idx].left = new_node(line);
575 return;
576 }
577 add_line_impl(nodes_[idx].left, l, mid, line);
578 return;
579 }
580
581 if (line.value_at(r) < nodes_[idx].line.value_at(r))
582 {
583 if (nodes_[idx].right == NIL)
584 {
585 nodes_[idx].right = new_node(line);
586 return;
587 }
588 add_line_impl(nodes_[idx].right, mid + 1, r, line);
589 }
590 }
591
592 [[nodiscard]] T query_impl(const size_t idx,
593 const T l, const T r,
594 const T x) const
595 {
596 using PromotedT = dp_optimization_detail::promoted_t<T>;
597 PromotedT ret = static_cast<PromotedT>(nodes_[idx].line.value_at(x));
598 if (l == r)
599 return static_cast<T>(ret);
600
601 const T mid = std::midpoint(l, r);
602 if (x <= mid and nodes_[idx].left != NIL)
603 ret = std::min(ret, static_cast<PromotedT>(query_impl(nodes_[idx].left, l, mid, x)));
604 else if (x > mid and nodes_[idx].right != NIL)
605 ret = std::min(ret, static_cast<PromotedT>(query_impl(nodes_[idx].right, mid + 1, r, x)));
606
607 return static_cast<T>(ret);
608 }
609
610 public:
611 Li_Chao_Tree(const T x_left, const T x_right)
612 : x_left_(x_left), x_right_(x_right)
613 {
614 ah_domain_error_if(x_left_ > x_right_)
615 << "Li_Chao_Tree: invalid domain [" << x_left_ << ", " << x_right_ << "]";
616 }
617
618 [[nodiscard]] bool is_empty() const noexcept { return root_ == NIL; }
619 [[nodiscard]] size_t node_count() const noexcept { return nodes_.size(); }
620
621 void clear()
622 {
623 Array<Node> tmp;
624 nodes_.swap(tmp);
625 root_ = NIL;
626 }
627
628 void add_line(const T slope, const T intercept)
629 {
630 const Line line{slope, intercept};
631 if (root_ == NIL)
632 {
633 root_ = new_node(line);
634 return;
635 }
636 add_line_impl(root_, x_left_, x_right_, line);
637 }
638
639 [[nodiscard]] T query(const T x) const
640 {
641 ah_runtime_error_if(root_ == NIL)
642 << "Li_Chao_Tree::query: no lines available";
643 ah_out_of_range_error_if(x < x_left_ or x > x_right_)
644 << "Li_Chao_Tree::query: x=" << x
645 << " outside domain [" << x_left_ << ", " << x_right_ << "]";
646
647 return query_impl(root_, x_left_, x_right_, x);
648 }

681 template <typename Cost>
682 [[nodiscard]] inline Monotone_Queue_DP_Result<Cost>
683 monotone_queue_min_dp(const Array<Cost> & base_cost,
684 const size_t window,
685 const Cost inf =
686 dp_optimization_detail::default_inf<Cost>())
687 {
688 const size_t n = base_cost.size();
689 if (n == 0)
690 return Monotone_Queue_DP_Result<Cost>{Array<Cost>(), Array<size_t>()};
691
692 ah_domain_error_if(window == 0 and n > 1)
693 << "monotone_queue_min_dp: window must be > 0 when n > 1";
694
695 Array<Cost> dp = Array<Cost>::create(n);
696 Array<size_t> parent = Array<size_t>::create(n);
697
698 Array<size_t> deque_idx;
699 size_t head = 0;
700
701 dp(0) = base_cost[0];
702 parent(0) = 0;
703 deque_idx.append(0);
704
705 for (size_t i = 1; i < n; ++i)
706 {
707 const size_t min_valid = (i > window) ? (i - window) : 0;
708
709 while (head < deque_idx.size() and deque_idx[head] < min_valid)
710 ++head;
711
712 ah_runtime_error_if(head == deque_idx.size())
713 << "monotone_queue_min_dp: no valid predecessor for i=" << i;
714
715 const size_t best = deque_idx[head];
716 const Cost bc = base_cost[i];
717 if (dp[best] >= inf or bc >= inf or dp[best] > inf - bc)
718 dp(i) = inf;
719 else
720 dp(i) = dp[best] + bc;
721
722 parent(i) = best;
723
724 while (deque_idx.size() > head and dp[deque_idx[deque_idx.size() - 1]] >= dp[i])
725 (void) deque_idx.remove_last();
726
727 deque_idx.append(i);
728 }
729
730 return Monotone_Queue_DP_Result<Cost>{std::move(dp), std::move(parent)};
731 }

754 template <typename T>
755 [[nodiscard]] inline Array<T>
756 min_weighted_squared_distance_1d(const Array<T> & xs,
757 const Array<T> & weights)
758 {
759 static_assert(std::is_integral_v<T> and std::is_signed_v<T>,
760 "min_weighted_squared_distance_1d requires signed integral T");
761
762 ah_domain_error_if(xs.size() != weights.size())
763 << "min_weighted_squared_distance_1d: xs and weights size mismatch";
764
765 const size_t n = xs.size();
766 if (n == 0)
767 return Array<T>();
768
769 T min_x = xs[0];
770 T max_x = xs[0];
771 for (size_t i = 1; i < n; ++i)
772 {
773 if (xs[i] < min_x)
774 min_x = xs[i];
775 if (xs[i] > max_x)
776 max_x = xs[i];
777 }
778
779 Li_Chao_Tree<T> lc(min_x, max_x);
780 using PromotedT = dp_optimization_detail::promoted_t<T>;
781 for (size_t j = 0; j < n; ++j)
782 {
783 const PromotedT px = static_cast<PromotedT>(xs[j]);
784 const T m = static_cast<T>(static_cast<PromotedT>(-2) * px);
785 const T b = static_cast<T>(px * px + static_cast<PromotedT>(weights[j]));
786 lc.add_line(m, b);
787 }
788
789 Array<T> ret = Array<T>::create(n);
790 for (size_t i = 0; i < n; ++i)
791 {
792 const PromotedT px = static_cast<PromotedT>(xs[i]);
793 ret(i) = static_cast<T>(px * px + static_cast<PromotedT>(lc.query(xs[i])));
794 }
795
796 return ret;
797 }
